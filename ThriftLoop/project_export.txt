@section Scripts {

    <script>
        function togglePassword(inputId) {
            const input = document.getElementById(inputId);
            input.type = input.type === 'password' ? 'text' : 'password';
        }

        document.getElementById('passwordInput').addEventListener('input', function () {
            const val  = this.value;
            const fill = document.getElementById('strengthFill');
            let strength = 0;
            if (val.length >= 8)          strength++;
            if (/[A-Z]/.test(val))        strength++;
            if (/[0-9]/.test(val))        strength++;
            if (/[^A-Za-z0-9]/.test(val)) strength++;

            const pct = (strength / 4) * 100;
            fill.style.width = pct + '%';
            fill.className   = 'strength-bar__fill';
            if (strength <= 1)      fill.classList.add('strength-bar__fill--weak');
            else if (strength <= 2) fill.classList.add('strength-bar__fill--fair');
            else if (strength <= 3) fill.classList.add('strength-bar__fill--good');
            else                    fill.classList.add('strength-bar__fill--strong');
        });
    </script>
}

    <script>
        (function () {
            'use strict';

            // ── Element references ─────────────────────────────────────────────
            var grid         = document.getElementById('itemGrid');
            var filterEmpty  = document.getElementById('filterEmpty');
            var visibleCount = document.getElementById('visibleCount');
            var searchInput  = document.getElementById('searchInput');
            var priceMin     = document.getElementById('priceMin');
            var priceMax     = document.getElementById('priceMax');
            var catChecks    = document.querySelectorAll('#catFilters .fy-check');
            var condChecks   = document.querySelectorAll('#condFilters .fy-check');

            // Guard — grid is absent on the global-empty state (no items at all).
            if (!grid) return;

            var cards = Array.prototype.slice.call(grid.querySelectorAll('.fy-card'));

            // ── Core filter function ───────────────────────────────────────────
            function applyFilters() {
                var query  = searchInput ? searchInput.value.trim().toLowerCase() : '';
                var minVal = priceMin    ? parseFloat(priceMin.value) || 0        : 0;
                var maxVal = priceMax    ? parseFloat(priceMax.value) || Infinity : Infinity;

                var activeCats = [];
                catChecks.forEach(function (cb) {
                    if (cb.checked) activeCats.push(cb.value);
                });

                var activeConds = [];
                condChecks.forEach(function (cb) {
                    if (cb.checked) activeConds.push(cb.value);
                });

                var shown = 0;

                cards.forEach(function (card) {
                    var cat   = card.dataset.category;
                    var cond  = card.dataset.condition;
                    var price = parseFloat(card.dataset.price);
                    var title = card.dataset.title || '';
                    var desc  = card.dataset.desc  || '';

                    var catOk   = activeCats.length  === 0 || activeCats.indexOf(cat)   !== -1;
                    var condOk  = activeConds.length === 0 || activeConds.indexOf(cond) !== -1;
                    var priceOk = price >= minVal && price <= maxVal;
                    var textOk  = !query || title.indexOf(query) !== -1 || desc.indexOf(query) !== -1;

                    var visible = catOk && condOk && priceOk && textOk;
                    card.classList.toggle('fy-card--hidden', !visible);
                    if (visible) shown++;
                });

                if (visibleCount) visibleCount.textContent = shown;

                if (filterEmpty) {
                    filterEmpty.classList.toggle('fy-empty--hidden', shown > 0);
                }
            }

            // ── Wire up all controls — live, no submit needed ──────────────────
            if (searchInput) searchInput.addEventListener('input',  applyFilters);
            if (priceMin)    priceMin.addEventListener('input',     applyFilters);
            if (priceMax)    priceMax.addEventListener('input',     applyFilters);
            catChecks.forEach(function (cb)  { cb.addEventListener('change', applyFilters); });
            condChecks.forEach(function (cb) { cb.addEventListener('change', applyFilters); });

            // ── Reset ──────────────────────────────────────────────────────────
            function resetAll() {
                if (searchInput) searchInput.value = '';
                if (priceMin)    priceMin.value    = priceMin.min  || 0;
                if (priceMax)    priceMax.value    = priceMax.max  || 100000;
                catChecks.forEach(function (cb)  { cb.checked = false; });
                condChecks.forEach(function (cb) { cb.checked = false; });
                applyFilters();
            }

            var resetBtn  = document.getElementById('resetFilters');
            var resetBtn2 = document.getElementById('resetFilters2');
            if (resetBtn)  resetBtn.addEventListener('click',  resetAll);
            if (resetBtn2) resetBtn2.addEventListener('click', resetAll);

            // ── Mobile sidebar toggle ──────────────────────────────────────────
            var toggleBtn = document.getElementById('fyFilterToggle');
            var sidebar   = document.getElementById('fySidebar');
            if (toggleBtn && sidebar) {
                toggleBtn.addEventListener('click', function () {
                    sidebar.classList.toggle('fy-sidebar--open');
                    toggleBtn.setAttribute('aria-expanded',
                        sidebar.classList.contains('fy-sidebar--open').toString());
                });
            }

        }());
    </script>
}

            function initCounter(inputId, counterId, max) {

                var counter = document.getElementById(counterId);
                if (!input || !counter) return;

                var update = function() {

                    counter.textContent = len + '/' + max;
                    counter.classList.toggle('create-field__counter--warn', len > max * 0.9);
                };

                input.addEventListener('input', update);
                update();
            }

            initCounter('titleInput', 'titleCounter', 100);
            initCounter('descInput', 'descCounter', 1000);

            // ── Category / Condition chips (required — click activates, cannot deselect) ──
            function initChips(containerId) {
                var container = document.getElementById(containerId);
                if (!container) return;

                container.addEventListener('click', function(e) {
                    var btn = e.target.closest('[data-value]');
                    if (!btn) return;

                    var select = document.getElementById(btn.dataset.target);
                    if (!select) return;

                    container.querySelectorAll('[data-value]').forEach(function(b) {
                        b.classList.remove('create-chip--active', 'create-condition__pill--active');
                    });

                    btn.classList.add(containerId === 'categoryChips' ? 'create-chip--active' : 'create-condition__pill--active');
                    select.value = btn.dataset.value;

                    if (window.$ && $.validator) $(select).valid();
                });
            }

            initChips('categoryChips');
            initChips('conditionPills');

            // ── Duration chips (required when Stealable — same chip pattern) ───
            function initDurationChips() {
                var container = document.getElementById('durationChips');
                var select    = document.getElementById('stealDurationSelect');
                if (!container || !select) return;

                container.addEventListener('click', function(e) {
                    var btn = e.target.closest('[data-value]');
                    if (!btn) return;

                    container.querySelectorAll('[data-value]').forEach(function(b) {
                        b.classList.remove('create-chip--active');
                    });

                    btn.classList.add('create-chip--active');
                    select.value = btn.dataset.value;

                    // Re-run unobtrusive validation on the hidden select so the
                    // error span clears immediately when the user picks a value.
                    if (window.$ && $.validator) $(select).valid();
                });
            }

            initDurationChips();

            // ── Size chips (optional — second click deselects) ─────────────────
            (function initSizeChips() {
                var container = document.getElementById('sizeChips');
                var select    = document.getElementById('sizeSelect');
                if (!container || !select) return;

                container.addEventListener('click', function(e) {
                    var btn = e.target.closest('[data-value]');
                    if (!btn) return;

                    var alreadyActive = btn.classList.contains('create-chip--active');

                    container.querySelectorAll('[data-value]').forEach(function(b) {
                        b.classList.remove('create-chip--active');
                    });

                    if (alreadyActive) {
                        select.value = '';
                    } else {
                        btn.classList.add('create-chip--active');
                        select.value = btn.dataset.value;

                });

            (function initStealableToggle() {
                var checkbox      = document.getElementById('IsStealable');
                var panel         = document.getElementById('stealDurationPanel');
                var durationSelect = document.getElementById('stealDurationSelect');
                if (!checkbox || !panel) return;

                function syncPanel() {
                    if (checkbox.checked) {
                        panel.removeAttribute('hidden');
                    } else {
                        panel.setAttribute('hidden', '');

                        // Clear the selection so the hidden field submits empty
                        // and server-side validation does not complain.
                        if (durationSelect) {
                            durationSelect.value = '';
                            document.querySelectorAll('#durationChips [data-value]').forEach(function(b) {
                                b.classList.remove('create-chip--active');
                            });
                        }
                    }
                }

                checkbox.addEventListener('change', syncPanel);
                // Run once on load to restore state after a failed server POST.
                syncPanel();
            })();

            // ── Photo dropzone ─────────────────────────────────────────────────
            var dropzone   = document.getElementById('photoDropzone');
            var input      = document.getElementById('photoInput');
            var empty      = document.getElementById('dropzoneEmpty');
            var preview    = document.getElementById('dropzonePreview');
            var previewImg = document.getElementById('previewImg');
            var removeBtn  = document.getElementById('removePhoto');
            var overlay    = document.getElementById('dropzoneOverlay');

            if (dropzone && input) {
                dropzone.addEventListener('click', function(e) {
                    if (e.target === removeBtn || (removeBtn && removeBtn.contains(e.target))) return;
                    input.click();
                });

                input.addEventListener('change', function() {
                    if (input.files && input.files[0]) showPreview(input.files[0]);
                });

                if (removeBtn) {
                    removeBtn.addEventListener('click', function(e) {
                        e.stopPropagation();
                        clearPreview();
                    });
                }

                ['dragenter', 'dragover'].forEach(function(evt) {
                    dropzone.addEventListener(evt, function(e) {

                ['dragleave', 'dragend', 'drop'].forEach(function(evt) {

                        e.preventDefault();

                        showDropzoneError('Only JPEG, PNG, or WebP images are accepted.');
                        return;
                    }
                    if (file.size > 5 * 1024 * 1024) {
                        showDropzoneError('Image must be smaller than 5 MB.');
                        return;
                    }

                    var dt = new DataTransfer();
                    dt.items.add(file);
                    input.files = dt.files;
                    showPreview(file);
                });
            }

            function showPreview(file) {
                var reader = new FileReader();
                reader.onload = function(e) {
                    previewImg.src = e.target.result;
                    if (empty)   empty.hidden   = true;
                    if (preview) preview.hidden = false;
                };
                reader.readAsDataURL(file);
            }

            function clearPreview() {
                input.value = '';
                previewImg.src = '';
                if (preview) preview.hidden = true;
                if (empty)   empty.hidden   = false;
            }

            function showDropzoneError(msg) {
                var errorSpan = document.querySelector('.create-photo-col .field__error');
                if (errorSpan) {
                    errorSpan.textContent = msg;
                    setTimeout(function() { errorSpan.textContent = ''; }, 3000);
                }
            }

            var submitBtn = document.getElementById('submitBtn');

                form.addEventListener('submit', function() {
                    submitBtn.disabled = true;
                    submitBtn.innerHTML = '<svg class="spin" viewBox="0 0 20 20" fill="currentColor" width="16" height="16"><path d="M10 3a7 7 0 100 14A7 7 0 0010 3zm-1 0v3M9 14v3M3 9H0M17 9h3"/></svg> Posting...';
                });
            }

            // ── Auto-dismiss toast ─────────────────────────────────────────────
            var toast = document.querySelector('.create-toast');
            if (toast) {
                setTimeout(function() {
                    toast.style.opacity = '0';
                    setTimeout(function() { toast.remove(); }, 400);
                }, 4000);
            }

        })();
    </script>

    <script>
        (function () {
            'use strict';

            // ── Generic countdown helper ─────────────────────────────────────
            // Accepts a <p> element with data-ends-at (ISO 8601 UTC string).
            // Updates its text every second. Stops when the deadline passes.
            function startCountdown(el) {
                if (!el) return;

                var endsAt = new Date(el.dataset.endsAt);

                // If the deadline has already passed when the page loads, do
                // nothing — the server-rendered Razor block already shows the
                // correct post-expiry state (finalise window / locked notice).
                // Without this guard a just-expired timer would call reload()
                // on the very first tick and cause an infinite refresh loop.
                if (endsAt <= new Date()) {
                    el.textContent = '00:00:00';
                    return;
                }

                var intervalId = null;

                function tick() {
                    var now    = new Date();
                    var diffMs = endsAt - now;

                    if (diffMs <= 0) {
                        el.textContent = '00:00:00';
                        // Stop ticking BEFORE reloading so the interval
                        // cannot fire again if the reload takes a moment.
                        clearInterval(intervalId);
                        // One-shot reload — the page re-renders with the
                        // correct Razor state for the new phase.
                        window.location.reload();
                        return;
                    }

                    var totalSecs = Math.floor(diffMs / 1000);
                    var h = Math.floor(totalSecs / 3600);
                    var m = Math.floor((totalSecs % 3600) / 60);
                    var s = totalSecs % 60;

                    el.textContent =
                        String(h).padStart(2, '0') + ':' +
                        String(m).padStart(2, '0') + ':' +
                        String(s).padStart(2, '0');
                }

                tick();
                intervalId = setInterval(tick, 1000);
            }

            // Start whichever timer is present on the page.
            startCountdown(document.getElementById('stealTimer'));
            startCountdown(document.getElementById('finalizeTimer'));

        }());
    </script>

    <script>
        (function() {
            'use strict';

            // ── Character counters ───────────────────────────────────────────────
            function initCounter(inputId, counterId, max) {
                const input = document.getElementById(inputId);
                const counter = document.getElementById(counterId);
                if (!input || !counter) return;

                const update = () => {
                    const len = input.value.length;
                    counter.textContent = len + '/' + max;
                    counter.classList.toggle('create-field__counter--warn', len > max * 0.9);
                };

                input.addEventListener('input', update);
                update();
            }

            initCounter('titleInput', 'titleCounter', 100);
            initCounter('descInput', 'descCounter', 1000);

            // ── Category / Condition chips (required — cannot deselect) ──────────
            function initChips(containerId) {
                const container = document.getElementById(containerId);
                if (!container) return;

                container.addEventListener('click', (e) => {
                    const btn = e.target.closest('[data-value]');
                    if (!btn) return;

                    const select = document.getElementById(btn.dataset.target);
                    if (!select) return;

                    container.querySelectorAll('[data-value]').forEach(b => {
                        b.classList.remove('create-chip--active', 'create-condition__pill--active');
                        b.setAttribute('aria-pressed', 'false');
                    });

                    btn.classList.add(containerId === 'categoryChips' ? 'create-chip--active' : 'create-condition__pill--active');
                    btn.setAttribute('aria-pressed', 'true');
                    select.value = btn.dataset.value;

                    if (window.$ && $.validator) $(select).valid();
                });

                container.addEventListener('keydown', (e) => {
                    if (e.key === ' ' || e.key === 'Enter') {
                        const btn = e.target.closest('[data-value]');
                        if (btn) { e.preventDefault(); btn.click(); }
                    }
                });
            }

            initChips('categoryChips');
            initChips('conditionPills');

            // ── Size chips (optional — clicking an active chip deselects it) ────────
            (function initSizeChips() {
                const container = document.getElementById('sizeChips');
                const select    = document.getElementById('sizeSelect');
                if (!container || !select) return;

                container.addEventListener('click', (e) => {
                    const btn = e.target.closest('[data-value]');
                    if (!btn) return;

                    const alreadyActive = btn.classList.contains('create-chip--active');

                    container.querySelectorAll('[data-value]').forEach(b => {
                        b.classList.remove('create-chip--active');
                        b.setAttribute('aria-pressed', 'false');
                    });

                    if (alreadyActive) {
                        // Second click on same chip → clear selection
                        select.value = '';
                    } else {
                        btn.classList.add('create-chip--active');
                        btn.setAttribute('aria-pressed', 'true');
                        select.value = btn.dataset.value;
                    }
                });
            })();

            // ── Photo dropzone ───────────────────────────────────────────────────
            const dropzone   = document.getElementById('photoDropzone');
            const input      = document.getElementById('photoInput');
            const emptyEl    = document.getElementById('dropzoneEmpty');
            const previewEl  = document.getElementById('dropzonePreview');
            const previewImg = document.getElementById('previewImg');
            const removeBtn  = document.getElementById('removePhoto');
            const overlay    = document.getElementById('dropzoneOverlay');
            const removeInput = document.getElementById('removeImageInput');

            if (dropzone && input) {
                dropzone.addEventListener('click', (e) => {
                    if (e.target === removeBtn || removeBtn?.contains(e.target)) return;
                    input.click();
                });

                input.addEventListener('change', () => {
                    if (input.files?.[0]) {
                        showPreview(input.files[0]);
                        removeInput.value = 'false';
                    }
                });

                removeBtn?.addEventListener('click', (e) => {
                    e.stopPropagation();
                    clearPreview();
                    removeInput.value = 'true';
                });

                ['dragenter', 'dragover'].forEach(evt => {
                    dropzone.addEventListener(evt, (e) => {
                        e.preventDefault();
                        e.stopPropagation();
                        overlay.hidden = false;
                        dropzone.classList.add('photo-dropzone--drag');
                    });
                });

                ['dragleave', 'dragend'].forEach(evt => {
                    dropzone.addEventListener(evt, (e) => {
                        e.preventDefault();
                        e.stopPropagation();
                        overlay.hidden = true;
                        dropzone.classList.remove('photo-dropzone--drag');
                    });
                });

                dropzone.addEventListener('drop', (e) => {
                    e.preventDefault();
                    e.stopPropagation();
                    overlay.hidden = true;
                    dropzone.classList.remove('photo-dropzone--drag');

                    const file = e.dataTransfer.files[0];
                    if (!file) return;

                    if (!['image/jpeg', 'image/png', 'image/webp'].includes(file.type)) {
                        showError('Only JPEG, PNG, or WebP images are accepted.');
                        return;
                    }
                    if (file.size > 5 * 1024 * 1024) {
                        showError('Image must be smaller than 5 MB.');
                        return;
                    }

                    const dt = new DataTransfer();
                    dt.items.add(file);
                    input.files = dt.files;
                    showPreview(file);
                    removeInput.value = 'false';
                });
            }

            function showPreview(file) {
                const reader = new FileReader();
                reader.onload = (e) => {
                    previewImg.src = e.target.result;
                    emptyEl.hidden = true;
                    previewEl.hidden = false;
                };
                reader.readAsDataURL(file);
            }

            function clearPreview() {
                input.value = '';
                previewImg.src = '';
                previewEl.hidden = true;
                emptyEl.hidden = false;
            }

            function showError(msg) {
                const errorSpan = document.querySelector('.create-photo-col .field__error');
                if (errorSpan) {
                    errorSpan.textContent = msg;
                    setTimeout(() => errorSpan.textContent = '', 3000);
                }
            }

            // ── Submit loading state ─────────────────────────────────────────────
            const form = document.getElementById('editForm');
            const submitBtn = document.getElementById('submitBtn');

            form?.addEventListener('submit', () => {
                submitBtn.disabled = true;
                submitBtn.innerHTML = '<svg class="spin" viewBox="0 0 20 20" fill="currentColor" width="16" height="16"><path d="M10 3a7 7 0 100 14A7 7 0 0010 3zm-1 0v3M9 14v3M3 9H0M17 9h3"/></svg> Saving...';
            });

        })();
    </script>

                this.setAttribute('aria-expanded', !isExpanded);
                dropdownMenu.classList.toggle('show');

                if (!dropdownToggle.contains(e.target) && !dropdownMenu.contains(e.target)) {
